void solve()
{
    int n, q;
    cin>>n>>q;
    vector<ll> a(n);
    for(int i = 0; i < n; i++)
        cin>>a[i];
    ll block_length = (int) sqrt(n + 0.0) + 1; 
    vector<ll> block(block_length, 0);
    for(int i = 0; i < n; i++)
        block[i / block_length] += a[i];
    for(int i = 0; i < q; i++)
    {
        int x;
        cin>>x;
        if(x == 1)
        {
            int k;
            ll u;
            cin>>k>>u;
            k--;
            block[k / block_length] -= a[k];
            a[k] = u;
            block[k / block_length] += u;
        }
        else
        {
            int l, r;
            cin>>l>>r;
            l--, r--;
            ll sum = 0;
            int i = l;
            while(i <= r)
            {
                if(i % block_length == 0 and i + block_length - 1 <= r)
                {
                    sum += block[i / block_length];
                    i += block_length;
                }
                else
                {
                    sum += a[i];
                    i++;
                }
            }
            cout<<sum<<endl;
        }
        
    }
}
